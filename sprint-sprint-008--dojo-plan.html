// services/api/src/routes/myst-ai.ts
import { Router, type Request, type Response } from 'express';
import { successResponse, errorResponse } from '...';
import type { AiBotCharacter, AiBotProfile } from '@tomachina/core';

const router = Router();

// GET /api/myst-ai - All character profiles
router.get('/', async (req: Request, res: Response) => {
  try {
    const profiles = getAllProfiles(); // static data
    return res.json(successResponse(profiles));
  } catch (err) {
    return res.status(500).json(errorResponse('Failed to load profiles'));
  }
});

// GET /api/myst-ai/:botName - Single bot bio page
router.get('/:botName', async (req: Request, res: Response) => {
  try {
    const { botName } = req.params;
    const profile = getProfileByName(botName);
    if (!profile) return res.status(404).json(errorResponse('Bot not found'));
    return res.json(successResponse(profile));
  } catch (err) {
    return res.status(500).json(errorResponse('Failed to load profile'));
  }
});

export default router;
Response format MUST be { success: boolean, data?: T, error?: string } per OS Rule #7. Use existing successResponse/errorResponse helpers from the API service.

  • Response format: { success: true, data: {...} }
