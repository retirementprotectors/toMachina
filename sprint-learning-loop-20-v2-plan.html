#!/bin/bash
# brain-save.sh - Auto-export warrior brain to dojo-warriors repo
set -euo pipefail

# 1. Detect warrior from active tmux session
WARRIOR=$(tmux display-message -p '#S' 2>/dev/null | tr '[:upper:]' '[:lower:]')
if [ -z "$WARRIOR" ]; then exit 0; fi

BRAIN_DIR="$HOME/Projects/dojo-warriors/$WARRIOR"
BRAIN_FILE="$BRAIN_DIR/brain.txt"
if [ ! -d "$BRAIN_DIR" ]; then exit 0; fi

# 2. Snapshot transcript
TRANSCRIPT=$(tmux capture-pane -p -S -1000 2>/dev/null)
if [ -z "$TRANSCRIPT" ]; then exit 0; fi

# 3. Redact secrets via regex
REDACTED=$(echo "$TRANSCRIPT" | sed -E \
  -e 's/(sk-[A-Za-z0-9_-]{20,})/[REDACTED_API_KEY]/g' \
  -e 's/(AIza[A-Za-z0-9_-]{30,})/[REDACTED_GCLOUD_KEY]/g' \
  -e 's/(password[=:][^ ]{5,})/password=[REDACTED]/gi' \
  -e 's/[0-9]{3}-[0-9]{2}-[0-9]{4}/[REDACTED_SSN]/g')

# 4. APPEND with session separator (idempotent via hash)
SESSION_HASH=$(echo "$REDACTED" | md5sum | cut -d' ' -f1)
if grep -q "hash:${SESSION_HASH}" "$BRAIN_FILE" 2>/dev/null; then exit 0; fi

printf "\n\n=== SESSION %s | hash:%s ===\n" "$(date -Iseconds)" "${SESSION_HASH}" >> "$BRAIN_FILE"
echo "$REDACTED" >> "$BRAIN_FILE"

# 5. Git commit + push
cd "$HOME/Projects/dojo-warriors"
git add "$WARRIOR/brain.txt"
git commit -m "brain-save: $WARRIOR $(date +%Y-%m-%d_%H:%M)"
git push origin main

function extractContent(message) {
  const content = message.message?.content;
  if (typeof content === 'string') {
    return { text: content, type: 'text', toolCalls: [] };
  }
  if (Array.isArray(content)) {
    const texts = content.filter(b => b.type === 'text').map(b => b.text);
    const tools = content.filter(b => b.type === 'tool_use');
    const type = tools.length > 0
      ? (texts.length > 0 ? 'mixed' : 'tool_only')
      : 'text';
    return { text: texts.join('\n'), type, toolCalls: tools };
  }
  return { text: '', type: 'text', toolCalls: [] };
}

// /home/jdm/mdj-agent/src/forge/phase-memory.ts
import { getFirestore, Timestamp } from 'firebase-admin/firestore';

export async function writePhaseMemory(
  sprintId: string, phase: string, discoveries: Discovery[]
): Promise<void> {
  const db = getFirestore();
  const batch = db.batch();
  for (const d of discoveries) {
    const ref = db.collection('knowledge_entries').doc();
    batch.set(ref, {
      id: ref.id, type: d.type, content: d.content,
      tags: [...d.tags, 'ronin', `sprint-${sprintId}`, `phase-${phase}`],
      confidence: d.confidence, machine: 'mdj1',
      source_session_id: sprintId,
      created_at: Timestamp.now(),
      promoted_to_claude_md: false,
      sprint_context: { sprint_id: sprintId, phase, tickets: d.relatedTickets }
    });
  }
  await batch.commit();
}
