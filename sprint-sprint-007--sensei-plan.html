export type SenseiTemplateType = 'workflow' | 'ui' | 'feat';

export interface SenseiContent {
  module_id: string;          // e.g. 'dex', 'rmd-center', 'households'
  title: string;              // "DEX Training Guide"

  screenshot_urls: string[];  // Firebase Storage paths: /training/{module}/{name}.png
  stats_query: string;        // Firestore collection path for live stats
  template_type: SenseiTemplateType;
  role_filter: string[];      // Entitlement keys: ['dex:read', 'admin']
  last_generated: Date;       // Timestamp of last HTML generation
  version: string;            // Semantic version: '1.0.0'
}

export interface SenseiAnalyticsEvent {
  event_type: 'train_response' | 'voltron_query' | 'popup_view';
  module_id: string;
  user_id: string;
  timestamp: Date;
  metadata?: Record<string, unknown>;
}

export interface SenseiGenerateResult {
  module_id: string;
  html_url: string;
  generated_at: Date;
  screenshot_count: number;
  version: string;
}

  { selector: '[data-module="notifications"]',module_id: 'notifications', label: 'Notifications' },
  { selector: '[data-module="mdj"]',          module_id: 'mdj',           label: 'MDJ' },

export function getRegistryEntry(moduleId: string): SenseiRegistryEntry | undefined {
  return SENSEI_REGISTRY.find(e => e.module_id === moduleId);

export function getRegisteredModuleIds(): string[] {
  return SENSEI_REGISTRY.map(e => e.module_id);
}
